  const nutrition_goals = await db.query('SELECT * FROM nutrition_goals WHERE client_id = $1 AND start_date <= $2 AND (end_date IS NULL OR end_date >= $2)', [clientId, weekStart]);
  
  if (!nutrition_goals[0]) return null; // –Ω–µ—Ç —Ü–µ–ª–µ–π

      total_protein,

      total_carbs,

  let proteinSum = 0, fatSum = 0, carbsSum = 0, fiberSum = 0, count = dailyData.length;
  
  dailyData.forEach(day => {

  return {
    protein: (proteinSum / count).toFixed(1),
    fat: (fatSum / count).toFixed(1),
    carbs: (carbsSum / count).toFixed(1),
    fiber: (fiberSum / count).toFixed(1)
